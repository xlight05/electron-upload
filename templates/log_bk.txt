            const getScanList = async () => {

                try {
                    const url = `${window.origin}/api/scan/list`
                    const scanListRequest = await fetch(url, {
                        method: "GET"
                    })
                    const scanList = await scanListRequest.json()
                    return scanList
                } catch (error) {
                    alert("Unable to load list")
                    return []
                }
            }

            const tableCreator = async () => {
                const tableData = await getScanList()

                let tableBody = ""
                tableData.map(data => {

                    tableBody += `
                                <tr>
                                    
                                    <td>${data.name}</td>
                                    <td>${data.size}</td>
                                    <td>${data.date}</td>
                                    <td><span class="${data.status == "Legitimate" ? "text-success" : "text-danger"}">${data.status}</span></td>
    
                                </tr>
                                `

                })

                document.querySelector("#scanTableData").innerHTML = tableBody

                $('#scanLogTable').DataTable();
            }

            window.addEventListener('DOMContentLoaded', (event) => {
                console.log('DOM fully loaded and parsed');

                tableCreator()

                console.log(`Doucment loaded`);

            });
